    <script> 
        const SERVER_URL = "COLE_A_URL_DO_NGROK_AQUI"; 
    </script>
    
    <script src="https://cdn.socket.io/4.7.5/socket.io.min.js"></script>
    <script>
        const socket = io(SERVER_URL);

        const messages = document.getElementById('messages');
        const form = document.getElementById('form');
        const input = document.getElementById('input');
        const statusDot = document.getElementById('status-dot');
        const statusText = document.getElementById('status-text');
        
        let activeChatId = null; 

        form.addEventListener('submit', function(e) {
            e.preventDefault();
            if (input.value && activeChatId) {
                socket.emit('reply from operator', { to: activeChatId, msg: input.value });

                const item = document.createElement('li');
                item.textContent = `VocÃª: ${input.value}`;
                item.style.fontWeight = 'bold';
                item.classList.add('text-right', 'bg-blue-100');
                messages.appendChild(item);
                messages.scrollTop = messages.scrollHeight;
                input.value = '';
            }
        });

        socket.on('new whatsapp message', function(msgData) {
            const item = document.createElement('li');
            item.textContent = `Cliente (${msgData.from}): ${msgData.msg}`;
            messages.appendChild(item);
            messages.scrollTop = messages.scrollHeight;
            
            activeChatId = msgData.from; 
        });

        socket.on('connect', () => {
            statusDot.classList.remove('bg-red-500');
            statusDot.classList.add('bg-green-500');
            statusText.textContent = 'Conectado';
        });

        socket.on('disconnect', () => {
            statusDot.classList.remove('bg-green-500');
            statusDot.classList.add('bg-red-500');
            statusText.textContent = 'Desconectado';
        });
    </script>
